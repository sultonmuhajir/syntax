Slice dan Splice


JavaScript
function xSlice(xx) {
   return xx.slice(2, 3);
}

function xSplice(xx) {
   return xx.splice(2, 3);
}

xSlice([1, 2, 3, 4, 5])   // [3]
xSplice([1, 2, 3, 4, 5])   // [3, 4, 5]


PHP
function xSplice($xx)
{
   return array_splice($xx, 2, 3);
}

xSplice([1, 2, 3, 4, 5])   // [3, 4, 5]


Python
def xSlice(xx):
   return xx[slice(2, 3)]

xSlice([1, 2, 3, 4, 5])   # [3]
