Min dan Max


JavaScript
function xMin(xx) {
   return Math.min(...xx);
}

function xMax(xx) {
   return Math.max(...xx);
}

xMin([1, 2, 3, 4, 5])   // 1
xMax([1, 2, 3, 4, 5])   // 5


PHP
function xMin($xx)
{
   return min($xx);
}

function xMax($xx)
{
   return max($xx);
}

xMin([1, 2, 3, 4, 5])   // 1
xMax([1, 2, 3, 4, 5])   // 5


Python
def xMin(xx):
   return min(xx)

def xMax(xx):
   return max(xx)

xMin([1, 2, 3, 4, 5])   # 1
xMax([1, 2, 3, 4, 5])   # 5
